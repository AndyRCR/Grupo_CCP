CREATE TABLE tipo_cliente (
    id_tipo_cliente       NUMBER PRIMARY KEY,
    nombre_tipo_cliente   VARCHAR2(20),
    estado                VARCHAR2(1)
);

CREATE TABLE tipo_empleado (
    id_tipo_empleado       NUMBER PRIMARY KEY,
    nombre_tipo_empleado   VARCHAR2(20),
    estado                 VARCHAR2(1)
);

CREATE TABLE tipo_comida (
    id_tipo_comida       NUMBER PRIMARY KEY,
    nombre_tipo_comida   VARCHAR2(20),
    estado               VARCHAR2(1)
);

CREATE TABLE comida (
    id_comida            NUMBER PRIMARY KEY,
    nombre_comida        VARCHAR2(20),
    descripcion_comida   VARCHAR2(60),
    precio_comida        VARCHAR2(45),
    id_tipo_comida       NUMBER
        REFERENCES tipo_comida,
    estado               VARCHAR2(1)
);

CREATE TABLE cliente (
    id_cliente        NUMBER PRIMARY KEY,
    nombre            VARCHAR2(20),
    apellido          VARCHAR2(20),
    sexo              VARCHAR2(10),
    dni               CHAR(8) NOT NULL UNIQUE,
    direccion         VARCHAR2(20),
    telefono          VARCHAR2(20),
    id_tipo_cliente   NUMBER
        REFERENCES tipo_cliente,
    estado            VARCHAR2(1)
);

CREATE TABLE empleado (
    id_empleado        NUMBER PRIMARY KEY,
    nombre             VARCHAR2(20),
    apellido           VARCHAR2(20),
    sexo               VARCHAR2(10),
    dni                CHAR(8) NOT NULL UNIQUE,
    direccion          VARCHAR2(20),
    telefono           VARCHAR2(20),
    id_tipo_empleado   NUMBER
        REFERENCES tipo_empleado,
    estado             VARCHAR2(1)
);

CREATE TABLE venta (
    id_venta      NUMBER PRIMARY KEY,
    fecha_venta   DATE,
    total_venta   NUMBER,
    id_empleado   NUMBER
        REFERENCES empleado,
    id_cliente    NUMBER
        REFERENCES cliente,
    estado        VARCHAR2(1)
);

CREATE TABLE detalle_venta (
    id_detalle_venta    NUMBER PRIMARY KEY,
    id_cantidad_venta   NUMBER,
    importe_venta       NUMBER,
    id_venta            NUMBER
        REFERENCES venta,
    id_comida           NUMBER
        REFERENCES comida,
    estado              VARCHAR2(1)
);
